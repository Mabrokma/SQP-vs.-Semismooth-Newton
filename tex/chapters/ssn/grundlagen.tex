

Wir betrachten erst mal die Definition
einer halbglatten Funktion.

\begin{definition}
\emph(Vgl. Definition 2.1 in \cite[S.~25]{ulbrich})\\
Sei $V \subset \R^n$ eine nichtleere offene Menge.
Sei $f \colon V \rightarrow \R^m$ lokal Lipschitz-stetig
in $x \in V$.
Sei $D_f$ die Menge aller Punkte,
wo $f$ eine (Fr\'echet-)Ableitung $f' \in L(\R^n,\R^m)$besitzt.
Die Menge
\begin{equation}
  \partial_B f(x) := \left\{ M \in \R^{m \times n}\ |\ 
    \exists (x_k) \subset D_f :
    x_k \rightarrow x, f'(x_k) \rightarrow M \right\}
\end{equation}
heißt B-Sub-Ableitung\footnote{englisch: B-subdifferential,
\textss{B} für Bouligand}
von $f$ an der Stelle $x$.
Ferner ist nach Clarke die allgemeine Jacobi-Matrix
von $f$ an der Stelle $x$ die konvexe Hülle
\begin{equation}
  \partial f(x) := \co \partial_B f(x).
\end{equation}
\end{definition}

\begin{definition}
\emph(Vgl. Definition 2.5 in \cite[S.~27]{ulbrich})\\
Sei $V \subset \R^n$ nichtleer und offen.
Die Funktion $f \colon V \rightarrow \R^m$ heißt halbglatt in $x \in V$,
falls $f$ lokal Lipschitz-stetig in $x$ ist und
\begin{equation}
  \lim_{M \in \partial f(x+ \tau d), d \rightarrow s, \tau \rightarrow 0^+} Md
\end{equation}
für alle $s \in \R^m$ existiert.

Falls $f$ halbglatt in allen $x \in V$ ist,
dann heißt $f$ halbglatt (in $V$).  
\end{definition}

Wir formulieren nun das halbglatte Newton-Verfahren
zum Lösen der Gleichung
\begin{equation}
  f(x) = 0, \label{eq:halbglatte_gleichung}
\end{equation}
wobei
$f \colon V \rightarrow \R^n$, $V \subset \R^n$ offen,
halbglatt an der Stelle der Lösung $\xopt \in V$ ist.

\begin{algorithm}\label{algo:halbglattes_newton}
\emph{(Das halbglatte Newton-Verfahren,
vgl. Algorithmus 2.11 in \cite[S.~29]{ulbrich})}
\begin{enumerate}
  \item Wähle einen Startpunkt $x^0$ und eine Abbruchschranke $\varepsilon > 0$.
        Setze $k := 0$.
  \item Ist $\|f(x^k)\| < \varepsilon$ \label{list:stop_criteria_ssn}
        $\Rightarrow$ STOP.
  \item Wähle $M_k \in \partial \nabla f (x^k)$
  \item Berechne die Lösung $d$ des linearen Gleichungssystems
        \begin{equation}
          M_k d = - f(x^k).
        \end{equation}
        Setze $d^k := d$.
  \item Setze $x^{k+1} := x^k + d^k$ und $k := k+1$ $\Rightarrow$
        Gehe zu Schritt~\ref{list:stop_criteria_ssn}.
\end{enumerate}
\end{algorithm}

Das halbglatte Newton-Verfahren arbeitet eigentlich genauso
wie das Newton-Verfahren.
Der Unterschied liegt nur darin, dass hier die Funktion nicht
unbedingt so glatt ist.

\begin{theorem}
\emph{(Vgl. Satz 2.12 in \cite[S.~29f]{ulbrich})}\\
Sei $f \colon V \rightarrow \R^n$ auf der offenen Menge $V \subset \R^n$
definiert.
Sei $\xopt$ eine Lösung des Problems~\eqref{eq:halbglatte_gleichung}.
Es gelte:
\begin{compactenum}[(i)]
  \item Die Funktion $f$ sei halbglatt in $\xopt$.
  \item Es existiere eine Konstante $C > 0$, sodass die Matrizen $M_k$
    für alle $k$ nichtsingulär mit $\| M_k^{-1} \| \leq C$ sind.
    \label{list:loesbarkeit_der_gleichung_in_halbgl_newton_verf}
\end{compactenum}
Dann existiert $\delta > 0$, sodass für alle
$x_0 \in U_{\delta}(\xopt)$
die Bedingung~(\ref{list:loesbarkeit_der_gleichung_in_halbgl_newton_verf})
erfüllt ist und das Verfahren~\ref{algo:halbglattes_newton}
entweder mit $x_k = \xopt$ terminiert
oder eine Folge $(x_k)$ erzeugt,
die $q$-superlinear gegen $\xopt$ konvergiert.
\end{theorem}
