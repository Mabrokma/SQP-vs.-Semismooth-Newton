
Sei $x^k$ ein zulässiger Punkt des Problems.
Wir approximieren die Funktion $f$ in der Umgebung von $x^k$.
\begin{equation}
  f(x) \approx f(x^k) + \nabla f(x^k)^T (x-x^k)
                 + \frac{1}{2} (x-x^k)^T f''(x^k) (x-x^k).
\end{equation}
Der Gradient von $f$ ist
\begin{equation}
  \nabla f(x) \approx \nabla f(x^k) + f''(x^k) x - f''(x^k) x^k.
\end{equation}

Die notwendige Bedingung für $x^{k+1}$ wäre
\begin{equation}
  \nabla f(x^{k+1}) + \lambda x^{k+1} + \mu^{k+1} = 0
\end{equation}
Ersetzen wir $\nabla f(x^{k+1})$ mit seiner Approximation,
bekommen wir die Gleichung
\begin{align}
  \nabla f(x^k) + f''(x^k) x^{k+1} - f''(x^k) x^k + \lambda x^{k+1} + \mu^{k+1}
    &= 0 \\
  f''(x^k) x^{k+1} + \lambda x^{k+1} + \mu^{k+1}
    &= f''(x^k) x^k - \nabla f(x^k)
\end{align}

\begin{algorithm}
\emph{(Aktive-Menge-Strategie)}
\begin{enumerate}
  \item $k := 0, x^k := x^0$
  \item Bestimme
        \begin{align}
          \A_+^k & := \{ i\,|\,x_i^k > b_i \} \\
          \A_-^k & := \{ i\,|\,x_i^k < a_i \} \\
          \I^k   & := \{ i\,|\,a_i \leq x_i^k \leq b_i \}
        \end{align}
  \item Löse das Problem
        \begin{align}
          f''(x^k) x^{k+1} + \lambda x^{k+1} + \mu^{k+1}
                      &= f''(x^k) x^k - \nabla f(x^k) \\
            x_i^{k+1} & = b_i \quad \text{ für } i \in \A_+^k \\
            x_i^{k+1} & = a_i \quad \text{ für } i \in \A_-^k \\
          \mu_i^{k+1} & = 0   \quad \text{ für } i \in \I^k
        \end{align}
  \item Falls
        \begin{align}
          0 & = \nabla f(x^{k+1}) + \lambda x^{k+1} + \mu^{k+1} \\
          \mu^{k+1} & = \max\left(0,\mu^{k+1}+\lambda(x^{k+1}-b)\right) +
                         \min\left(0,\mu^{k+1}+\lambda(x^{k+1}-a)\right)
        \end{align}
        $\Rightarrow$ STOP
  \item $x^k := x^{k+1}, k := k+1$ $\Rightarrow$ 2. %TODO use reference, not 2.
\end{enumerate}
\end{algorithm}
